function submitForm(e) {
  e.preventDefault();
  const f = e.target;
  const vorname   = f.querySelector('[name=vorname]').value.trim();
  const nachname  = f.querySelector('[name=nachname]').value.trim();
  const email     = f.querySelector('[name=email]').value.trim();
  const betreff   = f.querySelector('[name=betreff]').value;
  const nachricht = f.querySelector('[name=nachricht]').value.trim();

  const to      = 've@vivahome.de';
  const subject = encodeURIComponent((betreff || 'Kontaktanfrage') + ' – Viva Home GmbH');
  const body    = encodeURIComponent(
    'Von: ' + vorname + ' ' + nachname + '\n' +
    'E-Mail: ' + email + '\n' +
    'Betreff: ' + (betreff || '–') + '\n\n' +
    'Nachricht:\n' + nachricht
  );

  window.location.href = 'mailto:' + to + '?subject=' + subject + '&body=' + body;

  const msg = document.getElementById('form-msg');
  msg.style.display = 'block';
  msg.style.color = '#c8a96e';
  msg.textContent = '✓ Ihr E-Mail-Programm wird geöffnet – bitte senden Sie die vorbereitete Nachricht ab.';
  setTimeout(() => { msg.style.display = 'none'; }, 8000);
}
